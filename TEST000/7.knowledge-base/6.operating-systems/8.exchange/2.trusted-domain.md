# Domein dat je wilt controleren
$trustedDomain = "azurelogin-microsoft.com"

# Alle gebruikersmailboxen ophalen
$mailboxes = Get-Mailbox -RecipientTypeDetails UserMailbox -ResultSize Unlimited

# Controleren per mailbox
foreach ($mbx in $mailboxes) {
    $junkConfig = Get-MailboxJunkEmailConfiguration -Identity $mbx.Identity
    if ($junkConfig.TrustedSendersAndDomains -contains $trustedDomain) {
        Write-Host "✅ $($mbx.PrimarySmtpAddress): $trustedDomain is toegevoegd" -ForegroundColor Green
    } else {
        Write-Host "❌ $($mbx.PrimarySmtpAddress): $trustedDomain NIET gevonden" -ForegroundColor Red
    }
}
